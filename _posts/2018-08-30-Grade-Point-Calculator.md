<script>
    function _分2绩(x) {
        if (isNaN(x) || (x < 0) || (x > 100)) {
            return "出现错误";
        }
        if (x < 60) {
            return "0.000";
        }
        return (4 - 3 * (100 - x) * (100 - x) / 1600).toFixed(3);
    }

    function _绩2分(x) {
        if (!x) {
            return "[0,60)";
        }
        if (isNaN(x) || (x < 1) || (x > 4)) {
            return "出现错误";
        }
        return (100 - Math.sqrt(1600 * (4 - x) / 3)).toFixed(2);
    }
</script>
